 36 /**
 37  * Generates symbolic coordinates for the part of a construction including all the elements from that
 38  * a specific element depends of. These coordinates will be stored in GeometryElement.symbolic.
 39  * @param {JXG.Board} board The board that's element get some symbolic coordinates.
 40  * @param {JXG.GeometryElement} element All ancestor of this element get symbolic coordinates.
 41  * @param {String} variable Name for the coordinates, e.g. x or u.
 42  * @param {String} append Method for how to append the number of the coordinates. Possible values are
 43  *                        'underscore' (e.g. x_2), 'none' (e.g. x2), 'brace' (e.g. x[2]).
 44  * @type int
 45  * @return Number of coordinates given.
 46  */
 47 JXG.Math.Symbolic.generateSymbolicCoordinatesPartial = function(board, element, variable, append) {
 48     var makeCoords = function(num) {
 49         	if (append == 'underscore')
 50             	return '' + variable + '_{' + num + '}';
 51 	        else if (append == 'brace')
 52     	        return '' + variable + '[' + num + ']';
 53     	    else
 54     	        return '' + variable + '' + num;
 55 	    },
 56 	    list = element.ancestors,
 57 	    count = 0, t_num, t, k;
 58 
 59 	board.listOfFreePoints = [];
 60 	board.listOfDependantPoints = [];
 61 
 62     for(t in list) {
 63         t_num = 0;
 64         if (JXG.isPoint(list[t])) {
 65             for(k in list[t].ancestors) {
 66                 t_num++;
 67             }
 68             if(t_num == 0) {
 69                 list[t].symbolic.x = list[t].coords.usrCoords[1];
 70                 list[t].symbolic.y = list[t].coords.usrCoords[2];
 71                 board.listOfFreePoints.push(list[t]);
 72 
 73             } else {
 74                 count++;
 75                 list[t].symbolic.x = makeCoords(count);
 76                 count++;
 77                 list[t].symbolic.y = makeCoords(count);
 78                 board.listOfDependantPoints.push(list[t]);
 79             }
 80 
 81         }
 82     }
 83 
 84     if(JXG.isPoint(element)) {
 85         element.symbolic.x = 'x';
 86         element.symbolic.y = 'y';
 87     }
 88 
 89     return count;
 90 };

 92 /**
 93  * Clears all .symbolic.x and .symbolic.y members on every point of a given board.
 94  * @param {JXG.Board} board The board that's points get cleared their symbolic coordinates.
 95  */
 96 JXG.Math.Symbolic.clearSymbolicCoordinates = function(board) {
 97 	var clear = function(list) {
 98 		if((list == null) || (typeof list == 'undefined'))
 99 			return;
100 
101 		var t;
102 
103 	    for(t=0; t<list.length; t++) {
104 	        if (JXG.isPoint(list[t])) {
105 	            list[t].symbolic.x = '';
106 	            list[t].symbolic.y = '';
107 	        }
108 	    }
109 	};
110 
111 	clear(board.listOfFreePoints);
112 	clear(board.listOfDependantPoints);
113 
114     delete (board.listOfFreePoints);
115     delete (board.listOfDependantPoints);
116 };

118 /**
119  * Generates polynomials for the part of a construction including all the points from that
120  * a specific element depends of.
121  * @param {JXG.Board} board The board that's points polynomials will be generated.
122  * @param {JXG.GeometryElement} element All points in the set of ancestors of this element are used to generate the set of polynomials.
123  * @type Array
124  * @return Array of polynomials as strings.
125  */
126 JXG.Math.Symbolic.generatePolynomials = function(board, element, generateCoords) {
127     if(generateCoords)
128         this.generateSymbolicCoordinatesPartial(board, element, 'u', 'brace');
129 
130     var list = element.ancestors,
131         number_of_ancestors,
132         pgs = [],
133         result = [],
134         t, k, i;
135 
136     list[element.id] = element;
137 
138     for(t in list) {
139         number_of_ancestors = 0;
140         pgs = [];
141         if (JXG.isPoint(list[t])) {
142             for(k in list[t].ancestors) {
143                 number_of_ancestors++;
144             }
145             if(number_of_ancestors > 0) {
146                 pgs = list[t].generatePolynomial();
147                 for(i=0; i<pgs.length; i++)
148                     result.push(pgs[i]);
149             }
150         }
151     }
152 
153     if(generateCoords)
154         this.clearSymbolicCoordinates(board);
155 
156     return result;
157 };

159 /**
160  * Calculate geometric locus of a point given on a board. Invokes python script on server.
161  * @param {JXG.Board} board The board on that the point lies.
162  * @param {JXG.Point} point The point that will be traced.
163  * @param {function} callback A callback function that is called after the server request is finished.
164  *    Must take an array of strings as the only parameter.
165  * @type Array
166  * @return Array of points.
167  */
168 JXG.Math.Symbolic.geometricLocusByGroebnerBase = function(board, point, callback) {
169     var numDependent = this.generateSymbolicCoordinatesPartial(board, point, 'u', 'brace'),
170         poly, polyStr, result, oldRadius = {},
171         xsye = new JXG.Coords(JXG.COORDS_BY_USR, [0,0], board),
172         xeys = new JXG.Coords(JXG.COORDS_BY_USR, [board.canvasWidth, board.canvasHeight], board),
173         P1, P2, i, sf = 1, transx = 0, transy = 0, rot = 0, c, s, tx,
174         xs, xe, ys, ye,
175 
176         isIn = function(item, array) {
177         	var i, is = false;
178         	for(i=0; i<array.length; i++) {
179         		if(array[i].id == item) {
180         			is = true;
181         			break;
182         		}
183         	}
184         	return is;
185         },
186         bol = board.options.locus;
187 
188 
189     if(JXG.Server.modules.geoloci === JXG.undefined)
190         JXG.Server.loadModule('geoloci')
191 
192     if(JXG.Server.modules.geoloci === JXG.undefined)
193         throw new Error("JSXGraph: Unable to load JXG.Server module 'geoloci.py'.");
194 
195     xs = xsye.usrCoords[1];
196     xe = xeys.usrCoords[1];
197     ys = xeys.usrCoords[2];
198     ye = xsye.usrCoords[2];
199 
200     // Optimizations - but only if the user wants to
201     //   Step 1: Translate all related points, such that one point P1 (board.options.locus.toOrigin if set
202     //     or a random point otherwise) is moved to (0, 0)
203     //   Step 2: Rotate the construction around the new P1, such that another point P2 (board.options.locus.to10 if set
204     //     or a random point \neq P1 otherwise) is moved onto the positive x-axis
205     //  Step 3: Dilate the construction, such that P2 is moved to (1, 0)
206     //  Step 4: Give the scale factor (sf), the rotation (rot) and the translation vector (transx, transy) to
207     //    the server, which retransforms the plot (if any).
208 
209     // Step 1
210     if(bol.translateToOrigin && (board.listOfFreePoints.length > 0)) {
211 		if((bol.toOrigin !== JXG.undefined) && (bol.toOrigin != null) && isIn(bol.toOrigin.id, board.listOfFreePoints)) {
212 			P1 = bol.toOrigin;
213 		} else {
214 			P1 = board.listOfFreePoints[0];
215 		}
216 
217 		transx = P1.symbolic.x;
218 		transy = P1.symbolic.y;
219 		// translate the whole construction
220 		for(i=0; i<board.listOfFreePoints.length; i++) {
221 			board.listOfFreePoints[i].symbolic.x -= transx;
222 			board.listOfFreePoints[i].symbolic.y -= transy;
223 		}
224 
225         xs -= transx; xe -= transx;
226         ys -= transy; ye -= transy;
227 
228 		// Step 2
229 		if(bol.translateTo10 && (board.listOfFreePoints.length > 1)) {
230 			if((bol.to10 !== JXG.undefined) && (bol.to10 != null) && (bol.to10.id != bol.toOrigin.id) && isIn(bol.to10.id, board.listOfFreePoints)) {
231 				P2 = bol.to10;
232 			} else {
233 				if(board.listOfFreePoints[0].id == P1.id)
234 					P2 = board.listOfFreePoints[1]
235 				else
236 					P2 = board.listOfFreePoints[0];
237 			}
238 
239             rot = JXG.Math.Geometry.rad([1, 0], [0, 0], [P2.symbolic.x, P2.symbolic.y]);
240             c = Math.cos(-rot);
241             s = Math.sin(-rot);
242 
243 
244     		for(i=0; i<board.listOfFreePoints.length; i++) {
245                 tx = board.listOfFreePoints[i].symbolic.x;
246 		    	board.listOfFreePoints[i].symbolic.x = c*board.listOfFreePoints[i].symbolic.x - s*board.listOfFreePoints[i].symbolic.y;
247 	    		board.listOfFreePoints[i].symbolic.y = s*tx + c*board.listOfFreePoints[i].symbolic.y;
248     		}
249 
250             // thanks to the rotation this is zero
251             P2.symbolic.y = 0;
252 
253             tx = xs;
254             xs = c*xs - s*ys;
255             ys = s*tx + c*ys;
256             tx = xe;
257             xe = c*xe - s*ye;
258             ye = s*tx + c*ye;
259 
260 			// Step 3
261 			if(bol.stretch && (Math.abs(P2.symbolic.x) > JXG.Math.eps)) {
262                 sf = P2.symbolic.x;
263 
264  	    	    for(i=0; i<board.listOfFreePoints.length; i++) {
265   		    	    board.listOfFreePoints[i].symbolic.x /= sf;
266     		        board.listOfFreePoints[i].symbolic.y /= sf;
267    		        }
268 
269                 for(i in board.objects) {
270                     if((board.objects[i].elementClass == JXG.OBJECT_CLASS_CIRCLE) && (board.objects[i].method == 'pointRadius')) {
271                         oldRadius[i] = board.objects[i].radius;
272                         board.objects[i].radius /= sf;
273                     }
274                 }
275 
276                 xs /= sf; xe /= sf;
277                 ys /= sf; ye /= sf;
278 
279                 // this is now 1
280                 P2.symbolic.x = 1;
281             }
282 		}
283 
284         // make the coordinates "as rational as possible"
285         for(i=0; i<board.listOfFreePoints.length; i++) {
286             tx = board.listOfFreePoints[i].symbolic.x;
287             if(Math.abs(tx) < JXG.Math.eps)
288                 board.listOfFreePoints[i].symbolic.x = 0;
289             if(Math.abs(tx - Math.round(tx)) < JXG.Math.eps)
290                 board.listOfFreePoints[i].symbolic.x = Math.round(tx);
291 
292             tx = board.listOfFreePoints[i].symbolic.y;
293             if(Math.abs(tx) < JXG.Math.eps)
294                 board.listOfFreePoints[i].symbolic.y = 0;
295             if(Math.abs(tx - Math.round(tx)) < JXG.Math.eps)
296                 board.listOfFreePoints[i].symbolic.y = Math.round(tx);
297         }
298     }
299 
300     // end of optimizations
301 
302     poly = this.generatePolynomials(board, point);
303     polyStr = poly.join(',');
304 
305     this.cbp = function(data) {
306         //alert(data.exectime);
307         //callback(data.datax, data.datay, data.polynomial);
308         result = data;
309     };
310 
311     this.cb = JXG.bind(this.cbp, this);
312 
313     JXG.Server.modules.geoloci.lociCoCoA(xs, xe, ys, ye, numDependent, polyStr, sf, rot, transx, transy, this.cb, true);
314 
315     this.clearSymbolicCoordinates(board);
316 
317     for(i in oldRadius) {
318         board.objects[i].radius = oldRadius[i];
319     }
320 
321     return result;
322 };
