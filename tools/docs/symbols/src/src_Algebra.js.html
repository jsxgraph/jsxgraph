 31 /**
 32  * Creates a new instance of Algebra.
 33  * @class This class is just for backward compatibility and may be removed in future versions of JSXGraph. Hence,
 34  * please DO NOT USE ANY OF THESE METHODS but the given alternative.
 35  * @constructor
 36  */
 37 JXG.Algebra = function (/** JXG.Board */ board) {
 38     /**
 39      * Reference to board.
 40      * @type JXG.Board
 41      */
 42     this.board = board;
 43     
 44     /**
 45      * Defines float precision. Every number <tt>f</tt> with
 46      * Math.abs(f) < eps is assumed to be zero.
 47      * @default {@link JXG.Math#eps}
 48      * @see JXG.Math#eps
 49      */
 50     this.eps = JXG.Math.eps;
 51 };
