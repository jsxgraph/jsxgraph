 96     /**
 97             * Initialise a new board.
 98             * @param {String} box Html-ID to the Html-element in which the board is painted.
 99             * @return {JXG.Board} Reference to the created board.
100             */
101     this.initBoard = function (box, attributes) {
102         // Create a new renderer
103         var renderer,
104             originX, originY, unitX, unitY,
105             w, h, dimensions,
106             bbox,
107             zoomfactor, zoomX, zoomY,
108             showCopyright, showNavi,
109             board;
110 
111         dimensions = JXG.getDimensions(box);
112         if (typeof attributes == 'undefined') {
113             attributes = {};
114         }
115         if (typeof attributes["boundingbox"] != 'undefined') {
116             bbox = attributes["boundingbox"];
117             w = parseInt(dimensions.width);
118             h = parseInt(dimensions.height);
119             if (attributes["keepaspectratio"]) {
120             /**
121                                 * If the boundingbox attribute is given and the ratio of height and width of the sides defined by the bounding box and
122                                 * the ratio of the dimensions of the div tag which contains the board do not coincide,
123                                 * then the smaller side is chosen.
124                                 */
125                 unitX = w/(bbox[2]-bbox[0]);
126                 unitY = h/(-bbox[3]+bbox[1]);
127                 if (unitX<unitY) {
128                     unitY = unitX;
129                 } else {
130                     unitX = unitY;
131                 }
132             } else {
133                 unitX = w/(bbox[2]-bbox[0]);
134                 unitY = h/(-bbox[3]+bbox[1]);
135             }
136             originX = -unitX*bbox[0];
137             originY = unitY*bbox[1];
138         } else {
139             originX = ( (typeof attributes["originX"]) == 'undefined' ? 150 : attributes["originX"]);
140             originY = ( (typeof attributes["originY"]) == 'undefined' ? 150 : attributes["originY"]);
141             unitX = ( (typeof attributes["unitX"]) == 'undefined' ? 50 : attributes["unitX"]);
142             unitY = ( (typeof attributes["unitY"]) == 'undefined' ? 50 : attributes["unitY"]);
143         }
144         zoomfactor = ( (typeof attributes["zoom"]) == 'undefined' ? 1.0 : attributes["zoom"]);
145         zoomX = zoomfactor*( (typeof attributes["zoomX"]) == 'undefined' ? 1.0 : attributes["zoomX"]);
146         zoomY = zoomfactor*( (typeof attributes["zoomY"]) == 'undefined' ? 1.0 : attributes["zoomY"]);
147 
148         // ??? if (typeof attributes["showcopyright"] != 'undefined') attributes["showCopyright"] = attributes["showcopyright"];
149         showCopyright = ( (typeof attributes["showCopyright"]) == 'undefined' ? JXG.Options.showCopyright : attributes["showCopyright"]);
150 
151         if(JXG.Options.renderer == 'svg') {
152             renderer = new JXG.SVGRenderer(document.getElementById(box));
153         } else if(JXG.Options.renderer == 'vml') {
154             renderer = new JXG.VMLRenderer(document.getElementById(box));
155         } else {
156             renderer = new JXG.SilverlightRenderer(document.getElementById(box), dimensions.width, dimensions.height);
157         }
158 
159         board = new JXG.Board(box, renderer, '', [originX, originY], 1.0, 1.0, unitX, unitY, dimensions.width, dimensions.height,showCopyright);
160         this.boards[board.id] = board;
161         // board.initGeonextBoard();  // Contsruct "Ursprung" and other elements.
162         board.initInfobox();
163 
164         if((typeof attributes["axis"] != 'undefined') && attributes["axis"]) {
165         	board.defaultAxes = {};
166             board.defaultAxes.x = board.create('axis', [[0,0], [1,0]], {});
167             board.defaultAxes.y = board.create('axis', [[0,0], [0,1]], {});
168         }
169 
170         if ((typeof attributes["grid"] != 'undefined') && attributes["grid"]) {
171             board.renderer.drawGrid(board);
172         }
173 
174         if (typeof attributes["shownavigation"] != 'undefined') attributes["showNavigation"] = attributes["shownavigation"];
175         showNavi = ( (typeof attributes["showNavigation"]) == 'undefined' ? board.options.showNavigation : attributes["showNavigation"]);
176         if (showNavi) {
177             board.renderer.drawZoomBar(board);
178         }
179 
180         return board;
181     };

247     /**
248      * Free a board.
249      * @param {String} box Html-ID to the Html-element in which the board was painted.
250      */
251     this.freeBoard = function (board) {
252         var el;
253 
254         if(typeof(board) == 'string') {
255             board = this.boards[board];
256         }
257 
258         // Remove the event listeners
259         JXG.removeEvent(document, 'mousedown', board.mouseDownListener, board);
260         JXG.removeEvent(document, 'mouseup', board.mouseUpListener,board);
261         JXG.removeEvent(board.containerObj, 'mousemove', board.mouseMoveListener, board);
262 
263         // Remove all objects from the board.
264         for(el in board.objects) {
265             board.removeObject(board.objects[el]);
266         }
267 
268         // Remove all the other things, left on the board
269         board.containerObj.innerHTML = '';
270 
271         // Tell the browser the objects aren't needed anymore
272         for(el in board.objects) {
273             delete(board.objects[el]);
274         }
275 
276         // Free the renderer and the algebra object
277         delete(board.renderer);
278         delete(board.algebra);
279 
280         // Finally remove the board itself from the boards array
281         delete(this.boards[board.id]);
282     };

302 /**
303  * Parameter magic: object may be a string containing the name or id of the object or
304  * even the object itself, this function gets a returns to the object. Order: id/object, name.
305  * @param {JXG.Board} board Reference to the board the object belongs to.
306  * @param {String,Object} object String or reference to the object the reference is needed.
307  * @return {Object} Reference to the object given in parameter object
308  */
309 JXG.getReference = function(board, object) {
310     if(typeof(object) == 'string') {
311         if(board.objects[object] != null) { // Search by ID
312             object = board.objects[object];
313         } else if (board.elementsByName[object] != null) { // Search by name
314             object = board.elementsByName[object];
315         }
316     }
317 
318     return object;
319 };

371 /**
372   * Convert String, number or function to function.
373   * This method is used in Transformation.js
374   */
375 JXG.createEvalFunction = function(board,param,n) {
376     // convert GEONExT syntax into function
377     var f = [], i, str;
378 
379     for (i=0;i<n;i++) {
380         if (typeof param[i] == 'string') {
381             str = board.algebra.geonext2JS(param[i]);
382             str = str.replace(/this\.board\./g,'board.');
383             f[i] = new Function('','return ' + (str) + ';');
384         }
385     }
386     return function(k) {
387         var a = param[k];
388         if (typeof a == 'string') {
389             return f[k]();
390         } else if (typeof a=='function') {
391             return a();
392         } else if (typeof a=='number') {
393             return a;
394         }
395         return 0;
396     };
397 };

399 /**
400   * Convert String, number or function to function.
401   **/
402 JXG.createFunction = function(term,board,variableName,evalGeonext) {
403     var newTerm;
404 
405     if ((evalGeonext==null || evalGeonext==true) && JXG.isString(term)) {
406         // Convert GEONExT syntax into  JavaScript syntax
407         newTerm = board.algebra.geonext2JS(term);
408         return new Function(variableName,'return ' + newTerm + ';');
409     } else if (JXG.isFunction(term)) {
410         return term;
411     } else if (JXG.isNumber(term)) {
412         return function() { return term; };
413     } else if (JXG.isString(term)) {        // In case of string function like fontsize
414         return function() { return term; };
415     }
416     return null;
417 };

477 JXG.getDimensions = function(elementId) {
478     var element, display, els, originalVisibility, originalPosition,
479         originalDisplay, originalWidth, originalHeight;
480 
481     // Borrowed from prototype.js
482     element = document.getElementById(elementId);
483     if (element==null) {
484         throw new Error("\nJSXGraph: HTML container element '" + (elementId) + "' not found.");
485     }
486 
487     display = element.style['display'];
488     if (display != 'none' && display != null) {// Safari bug
489         return {width: element.offsetWidth, height: element.offsetHeight};
490     }
491 
492     // All *Width and *Height properties give 0 on elements with display none,
493     // so enable the element temporarily
494     els = element.style;
495     originalVisibility = els.visibility;
496     originalPosition = els.position;
497     originalDisplay = els.display;
498     els.visibility = 'hidden';
499     els.position = 'absolute';
500     els.display = 'block';
501 
502     originalWidth = element.clientWidth;
503     originalHeight = element.clientHeight;
504     els.display = originalDisplay;
505     els.position = originalPosition;
506     els.visibility = originalVisibility;
507     return {width: originalWidth, height: originalHeight};
508 };

543 /**
544   * getPosition: independent from prototype and jQuery
545   */
546 JXG.getPosition = function (Evt) {
547     var posx = 0,
548         posy = 0,
549         Evt;
550 
551     if (!Evt) {
552         Evt = window.event;
553     }
554 
555     if (Evt.pageX || Evt.pageY)     {
556         posx = Evt.pageX;
557         posy = Evt.pageY;
558     }
559     else if (Evt.clientX || Evt.clientY)    {
560         posx = Evt.clientX + document.body.scrollLeft + document.documentElement.scrollLeft;
561         posy = Evt.clientY + document.body.scrollTop + document.documentElement.scrollTop;
562     }
563     return [posx,posy];
564 };

569 JXG.getOffset = function (obj) {
570     var o=obj,
571         l=o.offsetLeft,
572         t=o.offsetTop;
573 
574     while(o=o.offsetParent) {
575         l+=o.offsetLeft;
576         t+=o.offsetTop;
577         if(o.offsetParent) {
578             l+=o.clientLeft;
579             t+=o.clientTop;
580         }
581     }
582     return [l,t];
583 };

648 /**
649  * Outputs a deep copy of an existing object and not only a flat copy.
650  * @param {Object} obj Object to be copied.
651  * @type Object
652  * @return Deep copy of given object.
653  */
654 JXG.deepCopy = function(obj) {
655     var c, i, prop, j;
656 
657     if (typeof obj !== 'object' || obj == null) {
658         return obj;
659     }
660     if (this.isArray(obj)) {
661         c = [];
662         for (i=0; i<obj.length; i++) {
663             prop = obj[i];
664             if (typeof prop == 'object') {
665                 if (this.isArray(prop)) {
666                     c[i] = [];
667                     for (j = 0; j < prop.length; j++) {
668                         if (typeof prop[j] != 'object') {
669                             c[i].push(prop[j]);
670                         } else {
671                             c[i].push(this.deepCopy(prop[j]));
672                         }
673                     }
674                 } else {
675                     c[i] = this.deepCopy(prop);
676                 }

681     } else {
682         c = {};
683         for (i in obj) {
684             prop = obj[i];
685             if (typeof prop == 'object') {
686                 if (this.isArray(prop)) {
687                     c[i] = [];

697                 }
698             } else {
699                 c[i] = prop;

701         }
702     }
