 34 /**
 35  * Converts expression of the form <i>leftop^rightop</i> into <i>Math.pow(leftop,rightop)</i>.
 36  * @param {String} te Expression of the form <i>leftop^rightop</i>
 37  * @type String
 38  * @return Converted expression.
 39  */
 40 JXG.GeonextParser.replacePow = function(te) {
 41     var count, pos, c,
 42         leftop, rightop, pre, p, left, i, right, expr;
 43     //te = te.replace(/\s+/g,''); // Loesche allen whitespace
 44                                 // Achtung: koennte bei Variablennamen mit Leerzeichen
 45                                 // zu Problemen fuehren.
 46     i = te.indexOf('^');
 47     while (i>=0) {
 48         left = te.slice(0,i);
 49         if (left.charAt(left.length-1)==')') {
 50             count = 1;
 51             pos = left.length-2;
 52             while (pos>=0 && count>0) {
 53                 c = left.charAt(pos);
 54                 if (c==')') { count++; }
 55                 else if (c=='(') { count--; }
 56                 pos--;
 57             }   
 58             if (count==0) {
 59                 leftop = '';
 60                 pre = left.substring(0,pos+1);   // finde evtl. F vor (...)^
 61                 p = pos;
 62                 while (p>=0 && pre.substr(p,1).match(/(\w+)/)) {
 63                     leftop = RegExp.$1+leftop;
 64                     p--;
 65                 }
 66                 leftop += left.substring(pos+1,left.length);
 67                 leftop = leftop.replace(/([\(\)\+\*\%\^\-\/\]\[])/g,"\\$1");
 68             }
 69         } else {
 70             leftop = '[\\w\\.]+';
 71         }
 72         right = te.slice(i+1);
 73         if (right.match(/^([\w\.]*\()/)) {
 74             count = 1;
 75             pos = RegExp.$1.length;
 76             while (pos<right.length && count>0) {
 77                 c = right.charAt(pos);
 78                 if (c==')') { count--; }
 79                 else if (c=='(') { count++; }
 80                 pos++;
 81             }
 82             if (count==0) {
 83                 rightop = right.substring(0,pos);
 84                 rightop = rightop.replace(/([\(\)\+\*\%\^\-\/\[\]])/g,"\\$1");
 85             }
 86         } else {
 87             rightop = '[\\w\\.]+';  // ^b 
 88         }
 89         expr = new RegExp('(' + leftop + ')\\^(' + rightop + ')');
 90         te = te.replace(expr,"JXG.Math.pow($1,$2)");
 91         i = te.indexOf('^');
 92     }
 93     return te;
 94 };

 96 /**
 97  * Converts expression of the form <i>If(a,b,c)</i> into <i>(a)?(b):(c)/i>.
 98  * @param {String} te Expression of the form <i>If(a,b,c)</i>
 99  * @type String
100  * @return Converted expression.
101  */
102 JXG.GeonextParser.replaceIf = function(te) {
103     var s = '',
104         left, right,
105         first = null,
106         second = null,
107         third = null,
108         i, pos, count, k1, k2, c, meat;
109     
110     i = te.indexOf('If(');
111     if (i<0) { return te; }
112 
113     te = te.replace(/""/g,'0'); // "" means not defined. Here, we replace it by 0
114     while (i>=0) {
115         left = te.slice(0,i);
116         right = te.slice(i+3); 
117         
118         // Search the end of the If() command and take out the meat
119         count = 1;
120         pos = 0;
121         k1 = -1;
122         k2 = -1;
123         while (pos<right.length && count>0) {
124             c = right.charAt(pos);
125             if (c==')') { 
126                 count--;
127             } else if (c=='(') {
128                 count++;
129             } else if (c==',' && count==1) {
130                 if (k1<0) { 
131                     k1 = pos; // first komma
132                 } else {
133                     k2 = pos; // second komma
134                 }
135             }
136             pos++;
137         } 
138         meat = right.slice(0,pos-1);
139         right = right.slice(pos);
140         
141         // Test the two kommas
142         if (k1<0) { return ''; } // , missing
143         if (k2<0) { return ''; } // , missing
144         
145         first = meat.slice(0,k1);
146         second = meat.slice(k1+1,k2);
147         third = meat.slice(k2+1);
148         first = this.replaceIf(first);    // Recurse
149         second = this.replaceIf(second);  // Recurse
150         third = this.replaceIf(third);    // Recurse
151 
152         s += left + '((' + first + ')?' + '('+second+'):('+third+'))';  
153         te = right;
154         first = null;
155         second = null;
156         i = te.indexOf('If(');
157     }
158     s += right;
159     return s;
160 };

162 /**
163  * Replace _{} by <sub>
164  * @param {String} te String containing _{}.
165  * @type String
166  * @return Given string with _{} replaced by <sub>.
167  */
168 JXG.GeonextParser.replaceSub = function(te) {
169     if(te['indexOf']) {} else return te;
170 
171     var i = te.indexOf('_{'),
172         j;
173     while (i>=0) {
174         te = te.substr(0,i)+te.substr(i).replace(/_\{/,'<sub>');
175         j = te.substr(i).indexOf('}');
176         if (j>=0) {
177             te = te.substr(0,j)+te.substr(j).replace(/\}/,'</sub>');
178         }
179         i = te.indexOf('_{');
180     }
181 
182     i = te.indexOf('_');
183     while (i>=0) {
184         te = te.substr(0,i)+te.substr(i).replace(/_(.?)/,'<sub>$1</sub>');
185         i = te.indexOf('_');
186     }
187     return te;
188 };

190 /**
191  * Replace ^{} by <sup>
192  * @param {String} te String containing ^{}.
193  * @type String
194  * @return Given string with ^{} replaced by <sup>.
195  */
196 JXG.GeonextParser.replaceSup = function(te) {
197     if(te['indexOf']) {} else return te;
198 
199     var i = te.indexOf('^{'),
200         j;
201     while (i>=0) {
202         te = te.substr(0,i)+te.substr(i).replace(/\^\{/,'<sup>');
203         j = te.substr(i).indexOf('}');
204         if (j>=0) {
205             te = te.substr(0,j)+te.substr(j).replace(/\}/,'</sup>');
206         }
207         i = te.indexOf('^{');
208     }
209 
210     i = te.indexOf('^');
211     while (i>=0) {
212         te = te.substr(0,i)+te.substr(i).replace(/\^(.?)/,'<sup>$1</sup>');
213         i = te.indexOf('^');
214     }
215 
216     return te;
217 };

219 /**
220  * Replace an element's name in terms by an element's id.
221  * @param term Term containing names of elements.
222  * @param board Reference to the board the elements are on.
223  * @return The same string with names replaced by ids.
224  **/
225 JXG.GeonextParser.replaceNameById = function(/** string */ term, /** JXG.Board */ board) /** string */ {
226     var pos = 0, end, elName, el, i,
227         funcs = ['X','Y','L','V'];
228     
229     for (i=0;i<funcs.length;i++) {
230         pos = term.indexOf(funcs[i]+'(');
231         while (pos>=0) {
232             if (pos>=0) {
233                 end = term.indexOf(')',pos+2);
234                 if (end>=0) {
235                     elName = term.slice(pos+2,end);
236                     elName = elName.replace(/\\(['"])?/g,"$1");
237                     el = board.elementsByName[elName];
238                     term = term.slice(0,pos+2) + el.id +  term.slice(end);
239                 }
240             }
241             end = term.indexOf(')',pos+2);
242             pos = term.indexOf(funcs[i]+'(',end);
243         }
244     }
245 
246     pos = term.indexOf('Dist(');
247     while (pos>=0) {
248         if (pos>=0) {
249             end = term.indexOf(',',pos+5);
250             if (end>=0) {
251                 elName = term.slice(pos+5,end);
252                 elName = elName.replace(/\\(['"])?/g,"$1");
253                 el = board.elementsByName[elName];
254                 term = term.slice(0,pos+5) + el.id +  term.slice(end);
255             }
256         }
257         end = term.indexOf(',',pos+5);
258         pos = term.indexOf(',',end);
259         end = term.indexOf(')',pos+1);
260         if (end>=0) {
261             elName = term.slice(pos+1,end);
262             elName = elName.replace(/\\(['"])?/g,"$1");
263             el = board.elementsByName[elName];
264             term = term.slice(0,pos+1) + el.id +  term.slice(end);
265         }
266         end = term.indexOf(')',pos+1);
267         pos = term.indexOf('Dist(',end);
268     }
269 
270     funcs = ['Deg','Rad'];
271     for (i=0;i<funcs.length;i++) {
272         pos = term.indexOf(funcs[i]+'(');
273         while (pos>=0) {
274             if (pos>=0) {
275                 end = term.indexOf(',',pos+4);
276                 if (end>=0) {
277                     elName = term.slice(pos+4,end);
278                     elName = elName.replace(/\\(['"])?/g,"$1");
279                     el = board.elementsByName[elName];
280                     term = term.slice(0,pos+4) + el.id +  term.slice(end);
281                 }
282             }
283             end = term.indexOf(',',pos+4);
284             pos = term.indexOf(',',end);
285             end = term.indexOf(',',pos+1);
286             if (end>=0) {
287                 elName = term.slice(pos+1,end);
288                 elName = elName.replace(/\\(['"])?/g,"$1");
289                 el = board.elementsByName[elName];
290                 term = term.slice(0,pos+1) + el.id +  term.slice(end);
291             }
292             end = term.indexOf(',',pos+1);
293             pos = term.indexOf(',',end);
294             end = term.indexOf(')',pos+1);
295             if (end>=0) {
296                 elName = term.slice(pos+1,end);
297                 elName = elName.replace(/\\(['"])?/g,"$1");
298                 el = board.elementsByName[elName];
299                 term = term.slice(0,pos+1) + el.id +  term.slice(end);
300             }
301             end = term.indexOf(')',pos+1);
302             pos = term.indexOf(funcs[i]+'(',end);
303         }
304     }
305     return term;
306 };

308 /**
309  * Replaces element ids in terms by element this.board.objects['id'].
310  * @param term A GEONE<sub>x</sub>T function string with JSXGraph ids in it.
311  * @return The input string with element ids replaced by this.board.objects["id"]. 
312  **/
313 JXG.GeonextParser.replaceIdByObj = function(/** string */ term) /** string */ {
314     var expr = /(X|Y|L)\(([\w_]+)\)/g;  // Suche "X(gi23)" oder "Y(gi23A)" und wandle in objects['gi23'].X() um.
315     term = term.replace(expr,"this.board.objects[\"$2\"].$1()");
316     
317     expr = /(V)\(([\w_]+)\)/g;  // Suche "X(gi23)" oder "Y(gi23A)" und wandle in objects['gi23'].X() um.
318     term = term.replace(expr,"this.board.objects[\"$2\"].Value()");
319 
320     expr = /(Dist)\(([\w_]+),([\w_]+)\)/g;  // 
321     term = term.replace(expr,'this.board.objects[\"$2\"].Dist(this.board.objects[\"$3\"])');
322 
323     expr = /(Deg)\(([\w_]+),([ \w\[\w_]+),([\w_]+)\)/g;  // 
324     term = term.replace(expr,'JXG.Math.Geometry.trueAngle(this.board.objects[\"$2\"],this.board.objects[\"$3\"],this.board.objects[\"$4\"])');
325 
326     expr = /Rad\(([\w_]+),([\w_]+),([\w_]+)\)/g;  // Suche Rad('gi23','gi24','gi25')
327     term = term.replace(expr,'JXG.Math.Geometry.rad(this.board.objects[\"$1\"],this.board.objects[\"$2\"],this.board.objects[\"$3\"])');
328     return term;
329 };

331 /**
332  * Converts the given algebraic expression in GEONE<sub>x</sub>T syntax into an equivalent expression in JavaScript syntax.
333  * @param {String} term Expression in GEONExT syntax
334  * @type String
335  * @return Given expression translated to JavaScript.
336  */
337 JXG.GeonextParser.geonext2JS = function(term, board) {
338     var expr, newterm, i,
339         from = ['Abs', 'ACos', 'ASin', 'ATan','Ceil','Cos','Exp','Floor','Log','Max','Min','Random','Round','Sin','Sqrt','Tan','Trunc'], 
340         to =   ['Math.abs', 'Math.acos', 'Math.asin', 'Math.atan', 'Math.ceil', 'Math.cos', 'Math.exp', 'Math.floor', 'Math.log', 'Math.max', 'Math.min', 'Math.random', 'this.board.round', 'Math.sin', 'Math.sqrt', 'Math.tan', 'Math.ceil'];
341     // removed: 'Pow'  -> Math.pow
342     
343     //term = JXG.unescapeHTML(term);  // This replaces > by >, < by < and & by &.ist aber zu allgemein
344     term = term.replace(/</g,'<'); // Hacks, to enable not well formed XML, @see JXG.GeonextReader#replaceLessThan
345     term = term.replace(/>/g,'>'); 
346     term = term.replace(/&/g,'&'); 
347     
348     // Umwandeln der GEONExT-Syntax in JavaScript-Syntax
349     newterm = term;
350     newterm = this.replaceNameById(newterm, board);
351     newterm = this.replaceIf(newterm);
352     // Exponentiations-Problem x^y -> Math(exp(x,y).
353     newterm = this.replacePow(newterm);
354     newterm = this.replaceIdByObj(newterm);
355     for (i=0; i<from.length; i++) {
356         expr = new RegExp(from[i],"ig");
357         newterm = newterm.replace(expr,to[i]);
358     }    
359     newterm = newterm.replace(/True/g,'true');
360     newterm = newterm.replace(/False/g,'false');
361     newterm = newterm.replace(/fasle/g,'false');
362 
363     newterm = newterm.replace(/Pi/g,'Math.PI');
364     return newterm;
365 };

367 /**
368  * Finds dependencies in a given term and resolves them by adding the
369  * dependent object to the found objects child elements.
370  * @param {JXG.GeometryElement} me Object depending on objects in given term.
371  * @param {String} term String containing dependencies for the given object.
372  * @param {JXG.Board} [board=me.board] Reference to a board
373  */
374 JXG.GeonextParser.findDependencies = function(me, term, board) {
375     if(typeof board=='undefined')
376         board = me.board;
377 
378     var elements = board.elementsByName,
379         el, expr, elmask;
380 
381     for (el in elements) {
382         if (el != me.name) {
383             if(elements[el].type == JXG.OBJECT_TYPE_TEXT) {
384                 if(!elements[el].isLabel) {
385                     elmask = el.replace(/\[/g,'\\[');
386                     elmask = elmask.replace(/\]/g,'\\]');
387                     expr = new RegExp("\\(\(\[\\w\\[\\]'_ \]+,\)*\("+elmask+"\)\(,\[\\w\\[\\]'_ \]+\)*\\)","g");  // Searches (A), (A,B),(A,B,C)
388                     if (term.search(expr)>=0) {
389                         elements[el].addChild(me);
390                     }
391                 }
392             }
393             else {
394                 elmask = el.replace(/\[/g,'\\[');
395                 elmask = elmask.replace(/\]/g,'\\]');
396                 expr = new RegExp("\\(\(\[\\w\\[\\]'_ \]+,\)*\("+elmask+"\)\(,\[\\w\\[\\]'_ \]+\)*\\)","g");  // Searches (A), (A,B),(A,B,C)
397                 if (term.search(expr)>=0) {
398                     elements[el].addChild(me);
399                 }
400             }
401         }
402     }
403 };
