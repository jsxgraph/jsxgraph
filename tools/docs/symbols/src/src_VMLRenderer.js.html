 32 JXG.VMLRenderer = function(container) {
 33     this.constructor();
 34     
 35     this.container = container;
 36     this.container.style.overflow = 'hidden';
 37     this.container.onselectstart = function () { return false; };
 38     
 39     this.resolution = 10; // Paths are drawn with a a resolution of this.resolution/pixel.
 40   
 41     // Add VML includes and namespace
 42     // Original: IE <=7
 43     container.ownerDocument.namespaces.add("jxgvml", "urn:schemas-microsoft-com:vml");
 44     //container.ownerDocument.createStyleSheet().addRule("v\\:*", "behavior: url(#default#VML);");
 45 
 46     this.container.ownerDocument.createStyleSheet().addRule(".jxgvml", "behavior:url(#default#VML)");
 47     try {
 48         !container.ownerDocument.namespaces.jxgvml && container.ownerDocument.namespaces.add("jxgvml", "urn:schemas-microsoft-com:vml");
 49         this.createNode = function (tagName) {
 50             return container.ownerDocument.createElement('<jxgvml:' + tagName + ' class="jxgvml">');
 51         };
 52     } catch (e) {
 53         this.createNode = function (tagName) {
 54             return container.ownerDocument.createElement('<' + tagName + ' xmlns="urn:schemas-microsoft.com:vml" class="jxgvml">');
 55         };
 56     }
 57         
 58     // um Dashes zu realisieren
 59     this.dashArray = ['Solid', '1 1', 'ShortDash', 'Dash', 'LongDash', 'ShortDashDot', 'LongDashDot'];    
 60 };

 76 JXG.VMLRenderer.prototype.setShadow = function(el) {
 77     var nodeShadow = el.rendNodeShadow;
 78     
 79     if (!nodeShadow) return;                          // Added 29.9.09. A.W.
 80     if (el.visPropOld['shadow']==el.visProp['shadow']) {
 81         return;
 82     }
 83     if(el.visProp['shadow']) {
 84         this.setAttr(nodeShadow, 'On', 'True');
 85         this.setAttr(nodeShadow, 'Offset', '3pt,3pt');
 86         this.setAttr(nodeShadow, 'Opacity', '60%');
 87         this.setAttr(nodeShadow, 'Color', '#aaaaaa');
 88     }
 89     else {
 90         this.setAttr(nodeShadow, 'On', 'False');
 91     }
 92     el.visPropOld['shadow']=el.visProp['shadow'];
 93 };

 95 JXG.VMLRenderer.prototype.setGradient = function(el) {
 96     var nodeFill = el.rendNodeFill;
 97     if(el.visProp['gradient'] == 'linear') {
 98         this.setAttr(nodeFill, 'type', 'gradient');
 99         this.setAttr(nodeFill, 'color2', el.visProp['gradientSecondColor']);
100         this.setAttr(nodeFill, 'opacity2', el.visProp['gradientSecondOpacity']);
101         this.setAttr(nodeFill, 'angle', el.visProp['gradientAngle']);
102     }
103     else if (el.visProp['gradient'] == 'radial') {
104         this.setAttr(nodeFill, 'type','gradientradial');
105         this.setAttr(nodeFill, 'color2',el.visProp['gradientSecondColor']);
106         this.setAttr(nodeFill, 'opacity2',el.visProp['gradientSecondOpacity']);
107         this.setAttr(nodeFill, 'focusposition', el.visProp['gradientPositionX']*100+'%,'+el.visProp['gradientPositionY']*100+'%');
108         this.setAttr(nodeFill, 'focussize', '0,0');
109     }
110     else {
111         this.setAttr(nodeFill, 'type','solid');
112     }
113 };

117 JXG.VMLRenderer.prototype.addShadowToGroup = function(groupname, board) {
118     var el, pEl;
119     if(groupname == "lines") {
120         for(el in board.objects) {
121             pEl = board.objects[el];
122             if(pEl.elementClass == JXG.OBJECT_CLASS_LINE) {
123                 this.addShadowToElement(pEl);
124             }
125         }
126     }
127     else if(groupname == "points") {
128         for(el in board.objects) {
129             pEl = board.objects[el];
130             if(pEl.elementClass == JXG.OBJECT_CLASS_POINT) {
131                 this.addShadowToElement(pEl);
132             }
133         }
134     }
135     else if(groupname == "circles") {
136         for(el in board.objects) {
137             pEl = board.objects[el];
138             if(pEl.elementClass == JXG.OBJECT_CLASS_CIRCLE) {
139                 this.addShadowToElement(pEl);
140             }
141         }
142     }    
143     board.fullUpdate();
144 };

198 JXG.VMLRenderer.prototype.updateTicks = function(axis,dxMaj,dyMaj,dxMin,dyMin) {
199     var tickArr = [], i, len, c, ticks, r = this.resolution;
200     
201     len = axis.ticks.length;
202     for (i=0; i<len; i++) {
203         c = axis.ticks[i].scrCoords;
204         if(axis.ticks[i].major) {
205             if (axis.labels[i].visProp['visible']) this.drawText(axis.labels[i]);        
206             tickArr.push(' m ' + Math.round(r*(c[1]+dxMaj)) + 
207                          ', ' + Math.round(r*(c[2]-dyMaj)) + 
208                          ' l ' + Math.round(r*(c[1]-dxMaj)) + 
209                          ', ' + Math.round(r*(c[2]+dyMaj))+' ');
210         }
211         else
212             tickArr.push(' m ' + Math.round(r*(c[1]+dxMin)) + 
213                          ', ' + Math.round(r*(c[2]-dyMin)) + 
214                          ' l ' + Math.round(r*(c[1]-dxMin)) + 
215                          ', ' + Math.round(r*(c[2]+dyMin))+' ');
216     }
217 
218     ticks = this.getElementById(axis.id);
219     if(ticks == null) {
220         ticks = this.createPrim('path', axis.id);
221         this.appendChildPrim(ticks,axis.layer);
222         this.appendNodesToElement(axis,'path');
223     } 
224     this.setAttr(ticks,'stroked', 'true');
225     this.setAttr(ticks,'strokecolor', axis.visProp['strokeColor'], 1);
226     this.setAttr(ticks,'strokeweight', axis.visProp['strokeWidth']);   
227     //ticks.setAttributeNS(null, 'stroke-opacity', axis.visProp['strokeOpacity']);
228     this.updatePathPrim(ticks, tickArr, axis.board);
229 };

247 JXG.VMLRenderer.prototype.transformImage = function(el,t) {
248     var node = el.rendNode, 
249         m;
250     m = this.joinTransforms(el,t);
251     node.style.left = (el.coords.scrCoords[1] + m[1][0]) + 'px'; 
252     node.style.top = (el.coords.scrCoords[2]-el.size[1] + m[2][0]) + 'px';    
253     node.filters.item(0).M11 = m[1][1];
254     node.filters.item(0).M12 = m[1][2];
255     node.filters.item(0).M21 = m[2][1];
256     node.filters.item(0).M22 = m[2][2];
257 };

259 JXG.VMLRenderer.prototype.joinTransforms = function(el,t) {
260     var m = [[1,0,0],[0,1,0],[0,0,1]], 
261         i,
262         len = t.length;
263         
264     for (i=0;i<len;i++) {
265         m = JXG.Math.matMatMult(t[i].matrix,m);
266     }
267     return m;
268 };

300 JXG.VMLRenderer.prototype.setObjectStrokeColor = function(el, color, opacity) {
301     var c = this.eval(color), 
302         o = this.eval(opacity), 
303         node, nodeStroke;
304 
305     o = (o>0)?o:0;
306 
307     if (el.visPropOld['strokeColor']==c && el.visPropOld['strokeOpacity']==o) {
308         return;
309     }
310     if(el.type == JXG.OBJECT_TYPE_TEXT) {
311         el.rendNode.style.color = c;
312     }        
313     else {       
314         node = el.rendNode;
315         this.setAttr(node,'stroked', 'true');
316         this.setAttr(node,'strokecolor', c);
317         
318         if(el.id == 'gridx') {
319             nodeStroke = this.getElementById('gridx_stroke');
320         }
321         else if(el.id == 'gridy') {
322             nodeStroke = this.getElementById('gridy_stroke');
323         }
324         else {
325             nodeStroke = el.rendNodeStroke;
326         }
327         if (o!=undefined) {
328             this.setAttr(nodeStroke,'opacity', (o*100)+'%');  
329             
330         }
331     }
332     el.visPropOld['strokeColor'] = c;
333     el.visPropOld['strokeOpacity'] = o;
334 };

412 JXG.VMLRenderer.prototype.createPrim = function(type, id) {
413     var node, 
414         fillNode = this.createNode('fill'), 
415         strokeNode = this.createNode('stroke'), 
416         shadowNode = this.createNode('shadow'), 
417         pathNode;
418     
419     this.setAttr(fillNode, 'id', this.container.id+'_'+id+'_fill');
420     this.setAttr(strokeNode, 'id', this.container.id+'_'+id+'_stroke');
421     this.setAttr(shadowNode, 'id', this.container.id+'_'+id+'_shadow');
422     
423     if (type=='circle' || type=='ellipse' ) {
424         node = this.createNode('oval');
425         node.appendChild(fillNode);
426         node.appendChild(strokeNode);
427         node.appendChild(shadowNode);
428     } else if (type == 'polygon' || type == 'path' || type == 'shape' || type == 'line') {    
429         node = this.createNode('shape');
430         node.appendChild(fillNode);
431         node.appendChild(strokeNode);
432         node.appendChild(shadowNode);   
433         pathNode = this.createNode('path');
434         this.setAttr(pathNode, 'id', this.container.id+'_'+id+'_path');        
435         node.appendChild(pathNode);
436     } else {
437         node = this.createNode(type);
438         node.appendChild(fillNode);
439         node.appendChild(strokeNode);
440         node.appendChild(shadowNode);
441     }
442     node.style.position = 'absolute';
443     this.setAttr(node, 'id', this.container.id+'_'+id);
444     
445     return node;
446 };

539 JXG.VMLRenderer.prototype.updatePathStringPrim = function(el) {
540     var pStr = [], 
541         //h = 3*el.board.canvasHeight, 
542         //w = 100*el.board.canvasWidth, 
543         i, scr,
544         r = this.resolution,
545         mround = Math.round,
546         symbm = ' m ', 
547         symbl = ' l ',
548         nextSymb = symbm, 
549         isNoPlot = (el.curveType!='plot'),
550         //isFunctionGraph = (el.curveType=='functiongraph'),
551         len = Math.min(el.numberPoints,8192); // otherwise IE 7 crashes in hilbert.html
552     
553     if (el.numberPoints<=0) { return ''; }
554     if (isNoPlot && el.board.options.curve.RDPsmoothing) {
555         el.points = this.RamenDouglasPeuker(el.points,1.0);
556     }
557     len = Math.min(len,el.points.length);
558 
559     for (i=0; i<len; i++) {
560         scr = el.points[i].scrCoords;
561         if (isNaN(scr[1]) || isNaN(scr[2]) /* || Math.abs(scr[1])>w || (isFunctionGraph && (scr[2]>h || scr[2]<-0.5*h))*/ ) {  // PenUp
562             nextSymb = symbm;
563         } else {
564             // IE has problems with values  being too far away.
565             if (scr[1]>20000.0) { scr[1] = 20000.0; }
566             else if (scr[1]<-20000.0) { scr[1] = -20000.0; }
567             if (scr[2]>20000.0) { scr[2] = 20000.0; }
568             else if (scr[2]<-20000.0) { scr[2] = -20000.0; }
569 
570             pStr.push([nextSymb,mround(r*scr[1]),', ',mround(r*scr[2])].join(''));
571             nextSymb = symbl;
572         }
573     }
574     pStr.push(' e');
575     return pStr;
576 };

578 JXG.VMLRenderer.prototype.updatePathStringPoint = function(el, size, type) {
579     var s = [],
580         scr = el.coords.scrCoords,
581         sqrt32 = size*Math.sqrt(3)*0.5,
582         s05 = size*0.5,
583         r = this.resolution;
584 
585     if(type == 'x') {
586         s.push(['m ',(r*(scr[1]-size)),', ',(r*(scr[2]-size)),' l ',
587         (r*(scr[1]+size)),', ',(r*(scr[2]+size)),' m ',
588         (r*(scr[1]+size)),', ',(r*(scr[2]-size)),' l ',
589         (r*(scr[1]-size)),', ',(r*(scr[2]+size))].join(''));
590     }
591     else if(type == '+') {
592         s.push(['m ',(r*(scr[1]-size)),', ',(r*(scr[2])),' l ',
593         (r*(scr[1]+size)),', ',(r*(scr[2])),' m ',
594         (r*(scr[1])),', ',(r*(scr[2]-size)),' l ',
595         (r*(scr[1])),', ',(r*(scr[2]+size))].join(''));    
596     }
597     else if(type == 'diamond') {
598         s.push(['m ',(r*(scr[1]-size)),', ',(r*(scr[2])),' l ',
599         (r*(scr[1])),', ',(r*(scr[2]+size)),' l ',
600         (r*(scr[1]+size)),', ',(r*(scr[2])),' l ',
601         (r*(scr[1])),', ',(r*(scr[2]-size)),' x e '
602         ].join(''));   
603     }
604     else if(type == 'A') {
605         s.push(['m ',(r*(scr[1])),', ',(r*(scr[2]-size)),' l ',
606         Math.round(r*(scr[1]-sqrt32)),', ',(r*(scr[2]+s05)),' l ',
607         Math.round(r*(scr[1]+sqrt32)),', ',(r*(scr[2]+s05)),' x e '
608         ].join(''));           
609     } 
610     else if(type == 'v') {
611         s.push(['m ',(r*(scr[1])),', ',(r*(scr[2]+size)),' l ',
612         Math.round(r*(scr[1]-sqrt32)),', ',(r*(scr[2]-s05)),' l ',
613         Math.round(r*(scr[1]+sqrt32)),', ',(r*(scr[2]-s05)),' x e '
614         ].join(''));       
615     }   
616     else if(type == '>') {
617         s.push(['m ',(r*(scr[1]+size)),', ',(r*(scr[2])),' l ',
618         (r*(scr[1]-s05)),', ',Math.round(r*(scr[2]-sqrt32)),' l ',
619         (r*(scr[1]-s05)),', ',Math.round(r*(scr[2]+sqrt32)),
620         //' x e '
621         ' l ',(r*(scr[1]+size)),', ',(r*(scr[2])) 
622         ].join(''));        
623     }
624     else if(type == '<') {
625         s.push(['m ',(r*(scr[1]-size)),', ',(r*(scr[2])),' l ',
626         (r*(scr[1]+s05)),', ',Math.round(r*(scr[2]-sqrt32)),' l ',
627         (r*(scr[1]+s05)),', ',Math.round(r*(scr[2]+sqrt32)),' x e '
628         ].join(''));    
629     }    
630     return s;
631 }

633 JXG.VMLRenderer.prototype.updatePolygonePrim = function(node,el) {
634     var minX = el.vertices[0].coords.scrCoords[1],
635         maxX = el.vertices[0].coords.scrCoords[1],
636         minY = el.vertices[0].coords.scrCoords[2],
637         maxY = el.vertices[0].coords.scrCoords[2],
638         i, 
639         len = el.vertices.length,
640         scr, x, y, 
641         pStr = [];
642         
643     this.setAttr(node, 'stroked', 'false');
644     for(i=1; i<len-1; i++) {
645         scr = el.vertices[i].coords.scrCoords;
646         if(scr[1] < minX) {
647             minX = scr[1];
648         }
649         else if(scr[1] > maxX) {
650             maxX = scr[1];
651         }
652         if(scr[2] < minY) {
653             minY = scr[2];
654         }
655         else if(scr[2] > maxY) {
656             maxY = scr[2];
657         }
658     }
659 
660     x = Math.round(maxX-minX); // Breite des umgebenden Rechtecks?
661     y = Math.round(maxY-minY); // Hoehe des umgebenden Rechtecks?
662 
663     if (!isNaN(x) && !isNaN(y)) {
664         node.style.width = x;
665         node.style.height = y;
666         this.setAttr(node, 'coordsize', x+','+y);
667     }
668      
669     scr = el.vertices[0].coords.scrCoords;
670     pStr.push(["m ",scr[1],",",scr[2]," l "].join(''));
671     
672     for(i=1; i<len-1; i++) {
673         scr = el.vertices[i].coords.scrCoords;
674         pStr.push(scr[1] + "," + scr[2]);
675         if(i<len-2) {
676             pStr.push(", ");
677         }
678     }
679     pStr.push(" x e");
680 
681     this.setAttr(node, 'path',pStr.join(""));
682 };

719 JXG.VMLRenderer.prototype.drawVerticalGrid = function(topLeft, bottomRight, gx, board) {
720     var node = this.createPrim('path', 'gridx'),
721         gridArr = [];
722         
723     while(topLeft.scrCoords[1] < bottomRight.scrCoords[1] + gx - 1) { 
724         gridArr.push(' m ' + (this.resolution*topLeft.scrCoords[1]) + 
725                      ', ' + 0 + 
726                      ' l ' + (this.resolution*topLeft.scrCoords[1]) + 
727                      ', ' + (this.resolution*board.canvasHeight)+' ');
728         topLeft.setCoordinates(JXG.COORDS_BY_SCREEN, [topLeft.scrCoords[1] + gx, topLeft.scrCoords[2]]);   
729     }
730     this.updatePathPrim(node, gridArr, board);
731     return node;
732 };

734 JXG.VMLRenderer.prototype.drawHorizontalGrid = function(topLeft, bottomRight, gy, board) {
735     var node = this.createPrim('path', 'gridy'),
736         gridArr = [];
737     while(topLeft.scrCoords[2] <= bottomRight.scrCoords[2] + gy - 1) {

739                      ', ' + (this.resolution*topLeft.scrCoords[2]) + 
740                      ' l ' + (this.resolution*board.canvasWidth) + 
741                      ', ' + (this.resolution*topLeft.scrCoords[2])+' ');
742         topLeft.setCoordinates(JXG.COORDS_BY_SCREEN, [topLeft.scrCoords[1], topLeft.scrCoords[2] + gy]);
743     }
744     this.updatePathPrim(node, gridArr, board);
745     return node;
