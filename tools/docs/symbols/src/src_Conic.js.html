 30 /**
 31  * @class This element is used to provide a constructor for an ellipse. An ellipse is given by two points (the foci) and a third point on the the ellipse or 
 32  * the length of the major axis.
 33  * @pseudo
 34  * @description
 35  * @name Ellipse
 36  * @augments JXG.Curve
 37  * @constructor
 38  * @type JXG.Curve
 39  * @throws {Exception} If the element cannot be constructed with the given parent objects an exception is thrown.
 40  * @param {JXG.Point,array_JXG.Point,array_JXG.Point,array} point1,point2,point3 Parent elements can be three elements either of type {@link JXG.Point} or array of
 41  * numbers describing the coordinates of a point. In the latter case the point will be constructed automatically as a fixed invisible point.
 42  * @param {JXG.Point,array_JXG.Point,array_number,function} point1,point2,number Parent elements can be two elements either of type {@link JXG.Point} or array of
 43  * numbers describing the coordinates of a point. The third parameter is a number/function which defines the length of the major axis
 44  * Optional parameters four and five are numbers which define the curve length (e.g. start/end). Default values are -pi and pi.
 45  * @example
 46  * // Create an Ellipse by three points
 47  * var A = board.create('point', [-1,4]);
 48  * var B = board.create('point', [-1,-4);
 49  * var C = board.create('point', [1,1]);
 50  * var el = board.create('ellipse',[A,B,C]);
 51  * </pre><div id="a4d7fb6f-8708-4e45-87f2-2379ae2bd2c0" style="width: 300px; height: 300px;"></div>
 52  * <script type="text/javascript">
 53  *   var glex1_board = JXG.JSXGraph.initBoard('a4d7fb6f-8708-4e45-87f2-2379ae2bd2c0', {boundingbox:[-6,6,6,-6], keepaspectratio:true, showcopyright: false, shownavigation: false});
 54  *   var A = glex1_board.create('point', [-1,4]);
 55  *   var B = glex1_board.create('point', [-1,-4);
 56  *   var C = glex1_board.create('point', [1,1]);
 57  *   var el = glex1_board.create('ellipse',[A,B,C]);
 58  * </script><pre>
 59  */
 60 JXG.createEllipse = function(board, parents, atts) {
 61     var F = [],  // focus 1 and focus 2
 62         C, majorAxis, i,
 63         rotationMatrix;
 64 
 65     atts = JXG.checkAttributes(atts,{withLabel:JXG.readOption(board.options,'conic','withLabel'), layer:null});
 66     // The foci and the third point are either points or coordinate arrays.
 67     for (i=0;i<2;i++) {
 68         if (parents[i].length>1) { // focus i given by coordinates
 69             F[i] = board.create('point', parents[i], {visible:false,fixed:true});
 70         } else if (JXG.isPoint(parents[i])) { // focus i given by point
 71             F[i] = JXG.getReference(board,parents[i]);
 72         } else if ((typeof parents[i] == 'function') && (parents[i]().elementClass == JXG.OBJECT_CLASS_POINT)) {  // given by function
 73             F[i] = parents[i]();
 74         } else if (JXG.isString(parents[i])) { // focus i given by point name
 75             F[i] = JXG.getReference(board,parents[i]);
 76         } else
 77             throw new Error("JSXGraph: Can't create Ellipse with parent types '" + (typeof parents[0]) + "' and '" + (typeof parents[1]) + "'.");
 78     }
 79     if (JXG.isNumber(parents[2])) { // length of major axis
 80         majorAxis = JXG.createFunction(parents[2],board);
 81     } else if ((typeof parents[2] == 'function') && (JXG.isNumber(parents[2]()))){
 82         majorAxis = parents[2];
 83     } else {
 84         if (JXG.isPoint(parents[2])) {                                               // point on ellipse
 85             C = JXG.getReference(board,parents[2]);
 86         } else if (parents[2].length>1) {                                            // point on ellipse given by coordinates
 87             C = board.create('point', parents[2], {visible:false,fixed:true});
 88         } else if ((typeof parents[2] == 'function') && (parents[2]().elementClass == JXG.OBJECT_CLASS_POINT)) {  // given by function
 89             C = parents[2]();
 90         } else if (JXG.isString(parents[2])) {                                      // focus i given by point name
 91             C = JXG.getReference(board,parents[2]);
 92         } else {
 93             throw new Error("JSXGraph: Can't create Ellipse with parent types '" + (typeof parents[0]) + "' and '" + (typeof parents[1]) + "' and '" + (typeof parents[2]) +"'.");
 94         }
 95         majorAxis = function(){ return C.Dist(F[0])+C.Dist(F[1]);};
 96     }
 97 
 98     if (typeof parents[4]=='undefined') parents[4] = 1.0001*Math.PI;   // to
 99     if (typeof parents[3]=='undefined') parents[3] = -1.0001*Math.PI;  // from
100 
101     atts = JXG.checkAttributes(atts,{curveType:'parameter'});
102 
103     var M = board.create('point', [
104                 function(){return (F[0].X()+F[1].X())*0.5;},
105                 function(){return (F[0].Y()+F[1].Y())*0.5;}
106             ],{visible:false, name:'', withLabel:false});
107 
108     var transformFunc = function() {
109             var ax = F[0].X(),
110                 ay = F[0].Y(),
111                 bx = F[1].X(),
112                 by = F[1].Y(),
113                 beta, co, si;
114 
115             // Rotate by the slope of the line [F[0],F[1]]
116             var sgn = (bx-ax>0)?1:-1;
117             if (Math.abs(bx-ax)>0.0000001) {
118                 beta = Math.atan2(by-ay,bx-ax)+ ((sgn<0)?Math.PI:0);
119             } else {
120                 beta = ((by-ay>0)?0.5:-0.5)*Math.PI;
121             }
122             co = Math.cos(beta);
123             si = Math.sin(beta);
124             var m = [
125                         [1,     0,  0],
126                         [M.X(),co,-si],
127                         [M.Y(),si, co]
128                     ];
129             return m;
130         };
131 
132     var curve = board.create('curve',[function(x) {return 0;},function(x) {return 0;},parents[3],parents[4]],atts);
133 
134     var polarForm = function(phi,suspendUpdate) {
135                 var a = majorAxis()*0.5,
136                     aa = a*a,
137                     e = F[1].Dist(F[0])*0.5,
138                     bb = aa-e*e,
139                     b = Math.sqrt(bb), 
140                     transformMat = [[1,0,0],[0,1,0],[0,0,1]],
141                     mx, my;
142                     
143                 if (!suspendUpdate) {
144                     rotationMatrix = transformFunc();
145                     mx = M.X();
146                     my = M.Y();
147                     transformMat[0][0] = rotationMatrix[0][0];
148                     transformMat[0][1] = 0.0;
149                     transformMat[0][2] = 0.0;
150                     transformMat[1][0] = mx*(1-rotationMatrix[1][1])+my*rotationMatrix[1][2];
151                     transformMat[1][1] = rotationMatrix[1][1];
152                     transformMat[1][2] = rotationMatrix[2][1];
153                     transformMat[2][0] = my*(1-rotationMatrix[1][1])-mx*rotationMatrix[1][2];
154                     transformMat[2][1] = rotationMatrix[1][2];
155                     transformMat[2][2] = rotationMatrix[2][2];
156                     curve.quadraticform = 
157                         JXG.Math.matMatMult(JXG.Math.Matrix.transpose(transformMat),
158                         JXG.Math.matMatMult(
159                             [
160                                 [-1+mx*mx/(a*a)+my*my/bb, -mx/aa , -mx/bb],
161                                 [-mx/aa                 ,   1/aa ,  0    ],
162                                 [-my/bb                 ,      0 ,  1/bb ]
163                             ],
164                         transformMat)); 
165                 }
166                 return JXG.Math.matVecMult(rotationMatrix,[1,a*Math.cos(phi),b*Math.sin(phi)]);
167         };
168 
169     curve.X = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[1];};
170     curve.Y = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[2];};
171     curve.midpoint = M;
172     curve.type = JXG.OBJECT_TYPE_CONIC;
173     return curve;
174 };

208         C,

222         } else if (JXG.isString(parents[i])) { // focus i given by point name

224         } else
225             throw new Error("JSXGraph: Can't create Hyperbola with parent types '" + (typeof parents[0]) + "' and '" + (typeof parents[1]) + "'.");
226     }
227     if (JXG.isNumber(parents[2])) { // length of major axis
228         majorAxis = JXG.createFunction(parents[2],board);
229     } else if ((typeof parents[2] == 'function') && (JXG.isNumber(parents[2]()))){
230         majorAxis = parents[2];
231     } else {
232         if (JXG.isPoint(parents[2])) {                                               // point on ellipse
233             C = JXG.getReference(board,parents[2]);
234         } else if (parents[2].length>1) {                                            // point on ellipse given by coordinates

238         } else if (JXG.isString(parents[2])) {                                      // focus i given by point name

240         } else {
241             throw new Error("JSXGraph: Can't create Hyperbola with parent types '" + (typeof parents[0]) + "' and '" + (typeof parents[1]) + "' and '" + (typeof parents[2]) +"'.");
242         }
243         majorAxis = function(){ return C.Dist(F[0])-C.Dist(F[1]);};
244     }
245 
246     if (typeof parents[4]=='undefined') parents[4] = 1.0001*Math.PI;   // to
247     if (typeof parents[3]=='undefined') parents[3] = -1.0001*Math.PI;  // from
248

251     var M = board.create('point', [
252                 function(){return (F[0].X()+F[1].X())*0.5;},

256     var transformFunc = function() {
257             var ax = F[0].X(),
258                 ay = F[0].Y(),
259                 bx = F[1].X(),
260                 by = F[1].Y(),

276         };

280     * Hyperbola is defined by (a*sec(t),b*tan(t)) and sec(t) = 1/cos(t)

283                 var a = majorAxis()*0.5,
284                     aa = a*a,
285                     e = F[1].Dist(F[0])*0.5,

290                     
291                 if (!suspendUpdate) {
292                     rotationMatrix = transformFunc();
293                     mx = M.X();
294                     my = M.Y();
295                     transformMat[0][0] = rotationMatrix[0][0];
296                     transformMat[0][1] = 0.0;
297                     transformMat[0][2] = 0.0;
298                     transformMat[1][0] = mx*(1-rotationMatrix[1][1])+my*rotationMatrix[1][2];

303                     transformMat[2][2] = rotationMatrix[2][2];
304                     curve.quadraticform = 
305                         JXG.Math.matMatMult(JXG.Math.Matrix.transpose(transformMat),
306                         JXG.Math.matMatMult(
307                             [
308                                 [-1+mx*mx/aa+my*my/bb, -mx/aa , my/bb],
309                                 [-mx/aa              ,    1/aa,  0   ],
310                                 [my/bb               ,      0 , -1/bb]
311                             ],
312                         transformMat)); 
313                 }
314                 return JXG.Math.matVecMult(rotationMatrix,[1,a/Math.cos(phi),b*Math.tan(phi)]);
315         };
316 
317     curve.X = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[1];};
318     curve.Y = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[2];};
319     curve.midpoint = M;
320     curve.type = JXG.OBJECT_TYPE_CONIC;
321     return curve;
322 };

324 /**
325  * @class This element is used to provide a constructor for a parabola. A parabola is given by one point (the focus) and a line (the directrix).
326  * @pseudo
327  * @description
328  * @name Parabola
329  * @augments JXG.Curve
330  * @constructor
331  * @type JXG.Curve
332  * @throws {Exception} If the element cannot be constructed with the given parent objects an exception is thrown.
333  * @param {JXG.Point,array_JXG.Line} point,line Parent elements are a point and a line.
334  * Optional parameters three and four are numbers which define the curve length (e.g. start/end). Default values are -pi and pi.
335  * @example
336  * // Create a parabola by a point C and a line l.
337  * var A = board.create('point', [-1,4]);
338  * var B = board.create('point', [-1,-4);
339  * var l = board.create('line', [A,B]);
340  * var C = board.create('point', [1,1]);
341  * var el = board.create('parabola',[C,l]);
342  * </pre><div id="524d1aae-217d-44d4-ac58-a19c7ab1de36" style="width: 300px; height: 300px;"></div>
343  * <script type="text/javascript">
344  *   var glex1_board = JXG.JSXGraph.initBoard('524d1aae-217d-44d4-ac58-a19c7ab1de36', {boundingbox:[-6,6,6,-6], keepaspectratio:true, showcopyright: false, shownavigation: false});
345  *   var A = glex1_board.create('point', [-1,4]);
346  *   var B = glex1_board.create('point', [-1,-4);
347  *   var l = glex1_board.create('line', [A,B]);
348  *   var C = glex1_board.create('point', [1,1]);
349  *   var el = glex1_board.create('parabola',[C,l]);
350  * </script><pre>
351  */
352 JXG.createParabola = function(board, parents, atts) {
353     var F1 = parents[0], // focus
354         l = parents[1],  // directrix
355         rotationMatrix;
356 
357     atts = JXG.checkAttributes(atts,{withLabel:JXG.readOption(board.options,'conic','withLabel'), layer:null});
358     if (parents[0].length>1) { // focus 1 given by coordinates
359         F1 = board.create('point', parents[0], {visible:false,fixed:true});
360     } else if (JXG.isPoint(parents[0])) { // focus i given by point
361         F1 = JXG.getReference(board,parents[0]);
362     } else if ((typeof parents[0] == 'function') && (parents[0]().elementClass == JXG.OBJECT_CLASS_POINT)) {  // given by function
363         F1 = parents[0]();
364     } else if (JXG.isString(parents[0])) { // focus i given by point name
365         F1 = JXG.getReference(board,parents[0]);
366     } else
367         throw new Error("JSXGraph: Can't create Parabola with parent types '" + (typeof parents[0]) + "' and '" + (typeof parents[1]) + "'.");
368 
369     if (typeof parents[3]=='undefined') parents[3] = 10.0;   // to
370     if (typeof parents[2]=='undefined') parents[2] = -10.0;  // from
371 
372     atts = JXG.checkAttributes(atts,{curveType:'parameter'});
373 
374     var M = board.create('point', [
375                 function() {
376                     var v = [0,l.stdform[1],l.stdform[2]];
377                     v = JXG.Math.crossProduct(v,F1.coords.usrCoords);
378                     return board.algebra.meetLineLine(v,l.stdform,0).usrCoords;
379                 }
380             ],{visible:false, name:'', withLabel:false});
381 
382     var transformFunc = function() {
383             var beta = Math.atan(l.getSlope()),                
384                 x = (M.X()+F1.X())*0.5,
385                 y = (M.Y()+F1.Y())*0.5;
386             beta += (F1.Y()-M.Y()<0 || (F1.Y()==M.Y() && F1.X()>M.X()) ) ? Math.PI : 0;
387 
388             // Rotate by the slope of the line l (Leitlinie = directrix)
389             var m = [
390                         [1,    0,             0],
391                         [x*(1-Math.cos(beta))+y*Math.sin(beta),Math.cos(beta),-Math.sin(beta)],
392                         [y*(1-Math.cos(beta))-x*Math.sin(beta),Math.sin(beta), Math.cos(beta)]
393                     ];
394             return m;
395         };
396 
397     var curve = board.create('curve',[function(x) {return 0;},function(x) {return 0;},parents[2],parents[3]],atts);
398 
399     var polarForm = function(t,suspendUpdate) {
400                 var e = M.Dist(F1)*0.5,
401                     transformMat = [[1,0,0],[0,1,0],[0,0,1]],
402                     a = (M.X()+F1.X())*0.5, 
403                     b = (M.Y()+F1.Y())*0.5;
404                 
405                 if (!suspendUpdate) {
406                     rotationMatrix = transformFunc();
407                     transformMat[0][0] = rotationMatrix[0][0];
408                     transformMat[0][1] = 0.0;
409                     transformMat[0][2] = 0.0;
410                     transformMat[1][0] = a*(1-rotationMatrix[1][1])+b*rotationMatrix[1][2];
411                     transformMat[1][1] = rotationMatrix[1][1];
412                     transformMat[1][2] = rotationMatrix[2][1];
413                     transformMat[2][0] = b*(1-rotationMatrix[1][1])-a*rotationMatrix[1][2];
414                     transformMat[2][1] = rotationMatrix[1][2];
415                     transformMat[2][2] = rotationMatrix[2][2];
416                     curve.quadraticform = 
417                         JXG.Math.matMatMult(JXG.Math.Matrix.transpose(transformMat),
418                         JXG.Math.matMatMult(
419                             [
420                                 [-b*4*e-a*a, a, 2*e],
421                                 [a,       -1, 0],
422                                 [2*e,      0, 0]
423                             ],
424                         transformMat)); 
425                 }
426                 return JXG.Math.matVecMult(rotationMatrix,[1,t+a,t*t/(e*4)+b]);
427         };
428     curve.X = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[1];};
429     curve.Y = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[2];};
430     curve.type = JXG.OBJECT_TYPE_CONIC;
431     return curve;
432 };

437  * @pseudo
438  * @description

456  *   var glex1_board = JXG.JSXGraph.initBoard('2d79bd6a-db9b-423c-9cba-2497f0b06320', {boundingbox:[-6,6,6,-6], keepaspectratio:true, showcopyright: false, shownavigation: false});

458  *   var B = glex1_board.create('point', [1,2]);
459  *   var C = glex1_board.create('point', [2,0]);
460  *   var D = glex1_board.create('point', [0,0]);
461  *   var E = glex1_board.create('point', [-1,5]);

465 JXG.createConic = function(board, parents, atts) {
466     var rotationMatrix = [[1,0,0],[0,1,0],[0,0,1]], 
467         eigen, a, b, c, M = [[1,0,0],[0,1,0],[0,0,1]],
468         c1, c2, points = [], i, definingMat, 
469         givenByPoints, 
470         p = [];
471 
472     if (parents.length==5) {
473         givenByPoints = true;
474     } else if (parents.length==6) {
475         givenByPoints = false;
476     } else 
477         throw new Error("JSXGraph: Can't create generic Conic with " + parent.length + " parameters.");  
478 
479     atts = JXG.checkAttributes(atts,{withLabel:JXG.readOption(board.options,'conic','withLabel'), layer:null});
480     if (givenByPoints) {
481         for (i=0;i<5;i++) {
482             if (parents[i].length>1) { // point i given by coordinates
483                 points[i] = board.create('point', parents[i], {visible:false,fixed:true});
484             } else if (JXG.isPoint(parents[i])) { // point i given by point
485                 points[i] = JXG.getReference(board,parents[i]);
486             } else if ((typeof parents[i] == 'function') && (parents[i]().elementClass == JXG.OBJECT_CLASS_POINT)) {  // given by function
487                 points[i] = parents[i]();
488             } else if (JXG.isString(parents[i])) { // point i given by point name
489                 points[i] = JXG.getReference(board,parents[i]);
490             } else
491                 throw new Error("JSXGraph: Can't create Conic section with parent types '" + (typeof parents[i]) + "'.");
492         }
493     } else {
494         /* Usual notation (x,y,z):
495          *  [[A0,A3,A4],
496          *   [A3,A1,A5],
497          *   [A4,A5,A2]]. 
498          * Our notation (z,x,y): 
499          *  [[-A2   , A4*2.0, A5*0.5],
500          *   [A4*2.0,    -A0, A3*0.5],
501          *   [A5*0.5, A3*0.5,    -A1]] 
502          * New: (z,x,y): 
503          *  [[A2, A4, A5],
504          *   [A4, A0, A3],
505          *   [A5, A3, A1]] 
506         */
507         definingMat = [[0,0,0],[0,0,0],[0,0,0]];
508         definingMat[0][0] = (JXG.isFunction(parents[2])) ? function(){ return    parents[2]();} : function(){ return    parents[2];};
509         definingMat[0][1] = (JXG.isFunction(parents[4])) ? function(){ return    parents[4]();} : function(){ return    parents[4];};
510         definingMat[0][2] = (JXG.isFunction(parents[5])) ? function(){ return    parents[5]();} : function(){ return    parents[5];};
511         definingMat[1][1] = (JXG.isFunction(parents[0])) ? function(){ return    parents[0]();} : function(){ return    parents[0];};
512         definingMat[1][2] = (JXG.isFunction(parents[3])) ? function(){ return    parents[3]();} : function(){ return    parents[3];};
513         definingMat[2][2] = (JXG.isFunction(parents[1])) ? function(){ return    parents[1]();} : function(){ return    parents[1];};
514     }
515 
516     // sym(A) = A + A^t . Manipulates A in place.
517     var sym = function(A) {
518         var i, j;
519         for (i=0;i<3;i++) {
520             for (j=i;j<3;j++) {
521                 A[i][j] += A[j][i];
522             }
523         }
524         for (i=0;i<3;i++) {
525             for (j=0;j<i;j++) {
526                 A[i][j] = A[j][i];
527             }
528         }
529         return A;
530     };
531 
532     // degconic(v,w) = sym(v*w^t)
533     var degconic = function(v,w) {
534         var i, j, mat = [[0,0,0],[0,0,0],[0,0,0]];
535         for (i=0;i<3;i++) {
536             for (j=0;j<3;j++) {
537                 mat[i][j] = v[i]*w[j];
538             }
539         }
540         return sym(mat);
541     };
542 
543     // (p^t*B*p)*A-(p^t*A*p)*B
544     var fitConic = function(A,B,p)  {
545         var pBp, pAp, Mv, mat = [[0,0,0],[0,0,0],[0,0,0]], i, j;
546         Mv = JXG.Math.matVecMult(B,p);
547         pBp = JXG.Math.innerProduct(p,Mv);
548         Mv = JXG.Math.matVecMult(A,p);
549         pAp = JXG.Math.innerProduct(p,Mv);
550         for (i=0;i<3;i++) {
551             for (j=0;j<3;j++) {
552                 mat[i][j] = pBp*A[i][j]-pAp*B[i][j];
553             }
554         }
555         return mat;
556     };
557  
558     // Here, the defining functions for the curve are just dummy functions.
559     // In polarForm there is a reference to curve.quadraticform.
560     var curve = board.create('curve',[function(x) {return 0;},function(x) {return 0;},0,2*Math.PI],atts);
561 
562     var polarForm = function(phi,suspendUpdate) {
563         var i, j, len, v;
564         if (!suspendUpdate) {
565             if (givenByPoints) {
566                 // Copy the point coordinate vectors
567                 for (i=0;i<5;i++) { 
568                     p[i] = points[i].coords.usrCoords; 
569                 }
570                 // Compute the quadratic form
571                 c1 = degconic(JXG.Math.crossProduct(p[0],p[1]),JXG.Math.crossProduct(p[2],p[3]));
572                 c2 = degconic(JXG.Math.crossProduct(p[0],p[2]),JXG.Math.crossProduct(p[1],p[3]));
573                 M = fitConic(c1,c2,p[4]);
574             } else {
575                 for (i=0;i<3;i++) {
576                     for (j=i;j<3;j++) {
577                         M[i][j] = definingMat[i][j]();
578                         if (j>i) M[j][i] = M[i][j];     
579                     }
580                 }
581             }
582             curve.quadraticform = M;    // Here is the reference back to the curve.
583             
584             // Compute Eigenvalues and Eigenvectors
585             eigen = JXG.Math.Numerics.Jacobi(M);
586             // Scale the Eigenvalues such that the first Eigenvalue is positive
587             if (eigen[0][0][0]<0) {
588                 eigen[0][0][0] *= (-1);
589                 eigen[0][1][1] *= (-1);
590                 eigen[0][2][2] *= (-1);
591             }
592             // Normalize the Eigenvectors
593             for (i=0;i<3;i++) {
594                 len = 0.0;
595                 for (j=0;j<3;j++) {
596                     len += eigen[1][j][i]*eigen[1][j][i];
597                 }
598                 len = Math.sqrt(len);
599                 for (j=0;j<3;j++) {
600                     //eigen[1][j][i] /= len;
601                 }
602             }
603             rotationMatrix = eigen[1];
604             //console.log(rotationMatrix);
605             //console.log(eigen[0]);
606             //console.log(c+' '+a+' '+b);
607             c = Math.sqrt(Math.abs(eigen[0][0][0]));
608             a = Math.sqrt(Math.abs(eigen[0][1][1]));
609             b = Math.sqrt(Math.abs(eigen[0][2][2]));
610 
611         }
612         if (eigen[0][1][1]<0.0 && eigen[0][2][2]<0.0) {
613             v = JXG.Math.matVecMult(rotationMatrix,[1/c,Math.cos(phi)/a,Math.sin(phi)/b]);
614         } else if (eigen[0][1][1]<0.0 && eigen[0][2][2]>0.0) {
615             v = JXG.Math.matVecMult(rotationMatrix,[Math.cos(phi)/c,1/a,Math.sin(phi)/b]);
616         } else if (eigen[0][2][2]<0.0) {
617             v = JXG.Math.matVecMult(rotationMatrix,[Math.sin(phi)/c,Math.cos(phi)/a,1/b]);
618         } 
619         // Normalize
620         v[1] /= v[0];
621         v[2] /= v[0];
622         v[0] = 1.0;
623         return v;
624     };
625 
626     curve.X = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[1];};
627     curve.Y = function(phi,suspendUpdate) {return polarForm(phi,suspendUpdate)[2];};
628 
629 
630     // Center coordinates see http://en.wikipedia.org/wiki/Matrix_representation_of_conic_sections
631     curve.midpoint = board.create('point',
632         [
633         function(){ 
634             var m = curve.quadraticform;
635             return [
636                 m[1][1]*m[2][2]-m[1][2]*m[1][2],
637                 m[1][2]*m[0][2]-m[2][2]*m[0][1],
638                 m[0][1]*m[1][2]-m[1][1]*m[0][2]
639             ];
640         }
641         ],{name:'',visible:false});
642 
643     curve.type = JXG.OBJECT_TYPE_CONIC;
644     return curve;
645 };

649 JXG.JSXGraph.registerElement('parabola', JXG.createParabola);
